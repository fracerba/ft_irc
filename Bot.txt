void Command::bot(std::vector<std::string> message, User &user, Server &server)  
{
    //std::cout << "Bot command " << message[2] << std::endl;
    std::string m = message[2];
    if (m.find(":") == 0)
        m = m.substr(1, m.size() - 1);
    std::string response = ":Solver PRIVMSG " + user.getNickname();
    if (m == "!what")
    {
        user.setError(0);
        response += " :What am I? I am The Solver of the Absolute Fabric, the Void, the Exponential End!\r\n";
	    send(user.getSocket(), response.c_str(), response.size(), 0);
    }
    else if (m == "!help")
    {
        user.setError(0);
        help(user);
        std::string response = ":Solver PRIVMSG " + user.getNickname() + " :With me you can use the following commands:\r\n";
        send(user.getSocket(), response.c_str(), response.size(), 0);
        response = ":Solver PRIVMSG " + user.getNickname() + " :!what - If you want to known what I am.\r\n";
        send(user.getSocket(), response.c_str(), response.size(), 0);
        response = ":Solver PRIVMSG " + user.getNickname() + " :!help - Print a list of all commands, including these.\r\n";
        send(user.getSocket(), response.c_str(), response.size(), 0);
        response = ":Solver PRIVMSG " + user.getNickname() + " :!users - Print a list of all the Users on the Server.\r\n";
        send(user.getSocket(), response.c_str(), response.size(), 0);
        response = ":Solver PRIVMSG " + user.getNickname() + " :!channels [all] - Print a list of all the joinable Channels on the Server, with the optional flag [all] Invite-only Channels will also be included.\r\n";
        send(user.getSocket(), response.c_str(), response.size(), 0);
        response = ":Solver PRIVMSG " + user.getNickname() + " :!restore <channel> - Set you as Operator in a channel that remained without one, obviously you need to be on that Channel.\r\n";
        send(user.getSocket(), response.c_str(), response.size(), 0);
        response = ":Solver PRIVMSG " + user.getNickname() + " :!quote - Print a random quote from a movie/TV series/videogame in italian, if you don't speak it, that's your fault.\r\n";
        send(user.getSocket(), response.c_str(), response.size(), 0);
    }
    else if (m == "!users")
    {
        //da fare
    }
    else if (m == "!channels")
    {
        //da fare
    }
    else if (m == "!restore")
    {
        //da fare
    }
    else if (m == "!quote")
    {
        user.setError(0);
        int r = rand() % 100;
        response += " :" + _qoutes[r] + "\r\n";
        send(user.getSocket(), response.c_str(), response.size(), 0);
    }
    else
    {
        user.setError(1 + user.getError());
        switch (user.getError())
        {
            case 1:
                response = ":Solver PRIVMSG " + user.getNickname() + " :Unknown command, please type a correct one.\r\n";
                break;
            case 2:
                response = ":Solver PRIVMSG " + user.getNickname() + " :Unknown command, please type a correct one.\r\n";
                break;
            case 3:
                response = ":Solver PRIVMSG " + user.getNickname() + " :Wrong command, you can check at the start of this conversation for the correct ones.\r\n";
                break;
            case 4:
                response = ":Solver PRIVMSG " + user.getNickname() + " :Wrong command, you can check at the start of this conversation for the correct ones.\r\n";
                break;
            case 5:
                response = ":Solver PRIVMSG " + user.getNickname() + " :Are you doing this on purpose? You're not gonna like it if you continue.\r\n";
                break;
            case 6:
                response = ":Solver PRIVMSG " + user.getNickname() + " :Has your brain started decaying? Can you even read?\r\n";
                break;
            case 7:
                response = ":Solver PRIVMSG " + user.getNickname() + " :Type a CORRECT command! This is the LAST warning.\r\n";
                break;
            case 8:
                response = ":Solver PRIVMSG " + user.getNickname() + " :DUMB MEAT PUPPET! I'LL ERASE YOU! I SWEAR!\r\n";
                break;
            default:
                response = ":Solver PRIVMSG " + user.getNickname() + " :I'm sorry, meat puppet, but you are too dumb to be here. Goodbye!\r\n";
                break;
        }
        send(user.getSocket(), response.c_str(), response.size(), 0);
        if (user.getError() >= 9)
        {
            std::vector<std::string> q;
            split("QUIT :Kicked by the Solver for being too dumb.", " ", q);
	    //messaggio di quit da fare per bene
            quit(q, user, server);
        }
    }
}
